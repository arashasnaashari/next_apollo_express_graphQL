[70] سوره معراج

بِسمِ اللّهِ الرَّحمنِ الرَّحيمِ

به نام خداوند بخشنده مهربان

 

 

 

1 سَأَلَ سائلُ بِعَذَاب وَاقِع

تقاضا كننده اى تقاضاى عذابى كرد كه انجام مى گيرد.(1)

 

2 لِّلْكفِرِينَ لَيْس لَهُ دَافِعٌ

(اين عذاب) براى كافران است، و هيچ كس نمى تواند آن را دفع كند، (2)

 

3 مِّنَ اللَّهِ ذِى الْمَعَارِجِ

از سوى خداوندى كه فرشتگانش بر آسمانها صعود مى كنند (3)

 

4 تَعْرُجُ الْمَلائِكَةُ وَ الرُّوحُ إِلَيْهِ فِي يَوْم كانَ مِقْدارُهُ خَمْسِينَ أَلْفَ سَنَة

فرشتگان و روح (آن فرشته مقرّب خداوند) به سوى او عروج مى كنند در روزى كه مقدارش پنجاه هزار سال است.(4)

 

5 فَاصْبِرْ صَبْراً جَمِيلاً

پس صبر جميل پيشه كن،(5)

 

6 إِنَّهُمْ يَرَوْنَهُ بَعِيداً

زيرا آنها آن روز را دور مى بينند،(6)

 

7 وَ نَراهُ قَرِيباً

و ما آن رانزديك مى بينيم.(7)

 

8 يَوْمَ تَكُونُ السَّماءُ كَالْمُهْلِ

 همان روزى كه آسمان همچون فلز گداخته مى شود،(8)

 

9 وَ تَكُونُ الْجِبالُ كَالْعِهْنِ

و كوهها مانند پشم رنگين (درفضا)متلاشى خواهدبود،(9)

 

10 وَ لا يَسْئَلُ حَمِيمٌ حَمِيماً

وهيچ دوستى صميمى سراغ دوستش رانمى گيرد!(10)

 

11 يُبَصَّرُونَهُمْ يَوَدُّ الْمُجْرِمُ لَوْ يَفْتَدِي مِنْ عَذابِ يَوْمِئِذ بِبَنِيهِ

آنها را به يكديگر نشان مى دهند (ولى هر كس گرفتار كار خويشتن است آن چنانكه) گنهكار دوست مى دارد فرزندان خود را در برابر عذاب آن روز فدا كند، (11)

 

12 وَ صاحِبَتِهِ وَ أَخِيهِ

و همسر و برادرش را،(12)

 

13 وَ فَصِيلَتِهِ الَّتِي تُؤْوِيهِ

و قبيله اش را كه از او حمايت مى كرد،(13)

 

14 وَ مَنْ فِي الأَرْضِ جَمِيعاً ثُمَّ يُنْجِيهِ

و همه مردم روى زمين را تا مايه نجاتش گردند;(14)

 

15 كَلاّ إِنَّها لَظى

هرگز چنين نيست (كه با اينها بتوان نجات يافت، آرى) تنها شعله هاى سوزان آتش است،(15)

 

16 نَزّاعَةً لِلشَّوى

كه پوست بدن را مى كند و مى بَرد.(16)

 

17 تَدْعُوا مَنْ أَدْبَرَ وَ تَوَلّى

و كسانى را كه (به فرمان خدا) پشت كردند (به سوى خود) مى خواند.(17)

 

18 وَ جَمَعَ فَأَوْعى

و (آنها كه اموال را) جمع و ذخيره كردند.(18)

 

19 إِنَّ الإِنْسانَ خُلِقَ هَلُوعاً

به يقين انسان حريص و كم طاقت آفريده شده است،(19)

 

20 إِذا مَسَّهُ الشَّرُّ جَزُوعاً

هنگامى كه بدى به او رسد بى تابى مى كند،(20)

 

21 وَ إِذا مَسَّهُ الْخَيْرُ مَنُوعاً

و هنگامى كه خوبى به او رسد مانع ديگران مى شود (و بخل مىورزد)، (21)

 

22 إِلاَّ الْمُصَلِّينَ

مگر نمازگزاران،(22)

 

23 الَّذِينَ هُمْ عَلى صَلاتِهِمْ دائِمُونَ

كسانى كه نمازهايشان را دائماً به جا مى آورند، (23)

 

24 وَ الَّذِينَ فِي أَمْوالِهِمْ حَقٌّ مَعْلُومٌ

و كسانى كه در اموالشان حق معيّنى است،(24)

 

25 لِلسّائِلِ وَ الْمَحْرُومِ

براى سائل و محروم،(25)

 

26 وَ الَّذِينَ يُصَدِّقُونَ بِيَوْمِ الدِّينِ

و آنها كه به روز جزا ايمان دارند،(26)

 

27 وَ الَّذِينَ هُمْ مِنْ عَذابِ رَبِّهِمْ مُشْفِقُونَ

و آنها كه از عذاب پروردگارشان بيمناكند،(27)

 

28 إِنَّ عَذابَ رَبِّهِمْ غَيْرُ مَأْمُون

چرا كه هيچ كس از عذاب پروردگارش در امان نيست،(28)

 

29 وَ الَّذِينَ هُمْ لِفُرُوجِهِمْ حافِظُونَ

و آنها كه دامان خويش را (از بى عفتى) حفظ مى كنند،(29)

 

30 إِلاّ عَلى أَزْواجِهِمْ أَوْ ما مَلَكَتْ أَيْمانُهُمْ فَإِنَّهُمْ غَيْرُ مَلُومِينَ

جز با همسران و كنيزان (كه در حكم همسرند آميزش ندارند)، به يقين چنين كسانى مورد سرزنش نخواهند بود. (30)

 

31 فَمَنِ ابْتَغى وَراءَ ذالِكَ فَأُولئِكَ هُمُ العادُونَ

و هر كس جز اينها را طلب كند، متجاوز است.(31)

 

32 وَ الَّذِينَ هُمْ لاِ َماناتِهِمْ وَ عَهْدِهِمْ راعُونَ

و آنها كه امانتها و پيمان خود را رعايت مى كنند،(32)

 

33 وَ الَّذِينَ هُمْ بِشَهاداتِهِمْ قائِمُونَ

و آنها كه به اداى شهادتشان قيام مى نمايند، (33)

 

34 وَ الَّذِينَ هُمْ عَلى صَلاتِهِمْ يُحافِظُونَ

و آنها كه بر نمازشان مواظبت دارند،(34)

 

35 أُولئِكَ فِي جَنّات مُكْرَمُونَ

آنان (كه چنين اوصافى دارند) در باغهاى بهشتى گرامى داشته مى شوند.(35)

 

36 فَما لِ الَّذِينَ كَفَرُوا قِبَلَكَ مُهْطِعِينَ

پس كافران را چه مى شود كه با سرعت نزد تو مى آيند،(36)

 

37 عَنِ الْيَمِينِ وَ عَنِ الشِّمالِ عِزِينَ

از راست و چپ، گروه گروه (و آرزوى بهشت دارند)؟!(37)

 

38 أَ يَطْمَعُ كُلُّ امْرِئ مِنْهُمْ أَنْ يُدْخَلَ جَنَّةَ نَعِيم

آيا هر يك از آنها (با اين اعمال زشتش) طمع دارد كه او را در بهشت پرنعمت الهى وارد كنند؟!(38)

 

39 كَلاّ إِنّا خَلَقْناهُمْ مِمّا يَعْلَمُونَ

هرگز چنين نيست; ما آنها را از آنچه خودشان مى دانند آفريده ايم.(39)

 

40 فَلا أُقْسِمُ بِرَبِّ الْمَشارِقِ وَ الْمَغارِبِ إِنّا لَقادِرُونَ

سوگند به پروردگار مشرقها و مغربها كه ما قادريم،(40)

 

41 عَلى أَنْ نُبَدِّلَ خَيْراً مِنْهُمْ وَ ما نَحْنُ بِمَسْبُوقِينَ

كه جاى آنان را به كسانى بدهيم كه از آنها بهترند; و ما هرگز مغلوب نخواهيم شد.(41)

 

42 فَذَرْهُمْ يَخُوضُوا وَ يَلْعَبُوا حَتّى يُلاقُوا يَوْمَهُمُ الَّذِي يُوعَدُونَ

آنان را به حال خود واگذار تا در باطل فرو روند و بازى كنند تا روز موعود خود را ملاقات نمايند.(42)

 

43 يَوْمَ يَخْرُجُونَ مِنَ الأَجْداثِ سِراعاً كَأَنَّهُمْ إِلى نُصُب يُوفِضُونَ

همان روز كه از قبرها بسرعت خارج مى شوند، گويى به سوى بتها مى دوند;(43)

 

44 خاشِعَةً أَبْصارُهُمْ تَرْهَقُهُمْ ذِلَّةٌ ذالِكَ الْيَوْمُ الَّذِي كانُوا يُوعَدُونَ

در حالى كه چشمهايشان (از شرم و وحشت) به زير افتاده، و (پرده اى از) ذلّت و خوارى آنها را پوشانده است. اين همان روزى است كه به آنها وعده داده مى شد.(44)

[71] سوره نوح

بِسمِ اللّهِ الرَّحمنِ الرَّحيمِ

به نام خداوند بخشنده مهربان

 

 

 

1 إِنّا أَرْسَلْنا نُوحاً إِلى قَوْمِهِ أَنْ أَنْذِرْ قَوْمَكَ مِنْ قَبْلِ أَنْ يَأْتِيَهُمْ عَذابٌ أَلِيمٌ

ما نوح را به سوى قومش فرستاديم و گفتيم: «قوم خود را انذار كن پيش از آن كه عذاب دردناكى به سراغشان آيد.»(1)

 

2 قالَ يا قَوْمِ إِنِّي لَكُمْ نَذِيرٌ مُبِينٌ

گفت: «اى قوم من! به يقين من براى شما انذار كننده آشكارى هستم،(2)

 

3 أَنِ اعْبُدُوا اللّهَ وَ اتَّقُوهُ وَ أَطِيعُونِ

كه خدا را پرستش كنيد و از مخالفت او بپرهيزيد و مرا اطاعت نماييد،(3)

 

4 يَغْفِرْ لَكُمْ مِنْ ذُنُوبِكُمْ وَ يُؤَخِّرْكُمْ إِلى أَجَل مُسَمّىً إِنَّ أَجَلَ اللّهِ إِذا جاءَ لا يُؤَخَّرُ لَوْ كُنْتُمْ تَعْلَمُونَ

تا خدا گناهانتان را بيامرزد و تا زمان معيّنى شما را عمر دهد; زيرا هنگامى كه اجل الهى فرا رسد، تأخيرى نخواهد داشت اگر مى دانستيد.»(4)

 

5 قالَ رَبِّ إِنِّي دَعَوْتُ قَوْمِي لَيْلاً وَ نَهاراً

(نوح) گفت: «پروردگارا! من قوم خود را شب و روز (به سوى تو) دعوت كردم،(5)

 

6 فَلَمْ يَزِدْهُمْ دُعائِي إِلاّ فِراراً

امّا دعوت من چيزى جز فرار (از حق) بر آنان نيفزود.(6)

 

7 وَ إِنِّي كُلَّما دَعَوْتُهُمْ لِتَغْفِرَ لَهُمْ جَعَلُوا أَصابِعَهُمْ فِي آذانِهِمْ وَ اسْتَغْشَوْا ثِيابَهُمْ وَ أَصَرُّوا وَ اسْتَكْبَرُوا اسْتِكْباراً

و من هر زمان آنها را دعوت كردم (كه ايمان بياورند) تا تو آنها را بيامرزى، انگشتان خويش را در گوشهايشان قرارداده و لباسهايشان را بر خود پيچيدند، و در مخالفت اصرار ورزيدند و بشدّت استكبار كردند.(7)

 

8 ثُمَّ إِنِّي دَعَوْتُهُمْ جِهاراً

سپس من آنها را با صداى بلند(به اطاعت فرمان تو) دعوت كردم،(8)

 

9 ثُمَّ إِنِّي أَعْلَنْتُ لَهُمْ وَ أَسْرَرْتُ لَهُمْ إِسْراراً

سپس آشكارا و نهان (حقيقت توحيد و ايمان را) براى آنان بيان داشتم.(9)

 

10 فَقُلْتُ اسْتَغْفِرُوا رَبَّكُمْ إِنَّهُ كانَ غَفّاراً

و گفتم: «از پروردگار خويش آمرزش بطلبيد كه او بسيار آمرزنده است،(10)

 

11 يُرْسِلِ السَّماءَ عَلَيْكُمْ مِدْراراً

تا بارانهاى پربركت آسمان را پى در پى بر شما فرو فرستد،(11)

 

12 وَ يُمْدِدْكُمْ بِأَمْوال وَ بَنِينَ وَ يَجْعَلْ لَكُمْ جَنّات وَ يَجْعَلْ لَكُمْ أَنْهاراً

و شما را با اموال و فرزندان فراوان كمك كند و باغهاى سرسبز و نهرهاى جارى در اختيارتان قرار دهد.(12)

 

13 ما لَكُمْ لا تَرْجُونَ لِلّهِ وَقاراً

چرا شما براى خدا عظمت قائل نيستيد؟!(13)

 

14 وَ قَدْ خَلَقَكُمْ أَطْواراً

در حالى كه شما را در مراحل مختلف آفريد (تا از نطفه به صورت انسان كامل رسيديد). (14)

 

15 أَ لَمْ تَرَوْا كَيْفَ خَلَقَ اللّهُ سَبْعَ سَماوات طِباقاً

آيا نمى دانيد چگونه خداوند هفت آسمان را برفراز يكديگر آفريده است، (15)

 

16 وَ جَعَلَ الْقَمَرَ فِيهِنَّ نُوراً وَ جَعَلَ الشَّمْسَ سِراجاً

و ماه را در ميان آن ها مايه روشنايى، و خورشيد را چراغ فروزانى قرارداده است؟!(16)

 

17 وَ اللّهُ أَنْبَتَكُمْ مِنَ الأَرْضِ نَباتاً

و خداوندشما را همچون گياهى از زمين رويانيد،(17)

 

18 ثُمَّ يُعِيدُكُمْ فِيها وَ يُخْرِجُكُمْ إِخْراجاً

سپس شما را به زمين باز مى گرداند، و بار ديگر شما را خارج مى سازد.(18)

 

19 وَ اللّهُ جَعَلَ لَكُمُ الأَرْضَ بِساطاً

و خداوند زمين را براى شما فرش گسترده اى قرار داد،(19)

 

20 لِتَسْلُكُوا مِنْها سُبُلاً فِجاجاً

تا از راههاى وسيع و درّه هاى آن بگذريد (و به هر جا مى خواهيد برويد).»(20)

 

21 قالَ نُوحٌ رَبِّ إِنَّهُمْ عَصَوْنِي وَ اتَّبَعُوا مَنْ لَمْ يَزِدْهُ مالُهُ وَ وَلَدُهُ إِلاّ خَساراً

نوح (بعد از نوميدى از هدايت آنان) گفت: «پروردگارا! آنها نافرمانى من كردند و از كسانى پيروى نمودند كه اموال و فرزندانشان چيزى جز زيانكارى بر آنها نيفزوده است.(21)

 

22 وَ مَكَرُوا مَكْراً كُبّاراً

و (اين رهبران گمراه) مكر عظيمى به كار بردند،(22)

 

23 وَ قالُوا لا تَذَرُنَّ آلِهَتَكُمْ وَ لا تَذَرُنَّ وَدّاً وَ لا سُواعاً وَ لا يَغُوثَ وَ يَعُوقَ وَ نَسْراً

و گفتند: دست از معبودان خود برنداريد (بخصوص) بتهاى «ودّ»، «سُواع»، «يَغُوث»، «يعوق» و «نسر» را رها نكنيد. (23)

 

24 وَ قَدْ أَضَلُّوا كَثِيراً وَ لا تَزِدِ الظّالِمِينَ إِلاّ ضَلالاً

و آنها بسيارى را گمراه كردند. خداوندا، ستمكاران را جز گمراهى ميفزا!» (24)

 

25 مِمّا خَطِيئاتِهِمْ أُغْرِقُوا فَأُدْخِلُوا ناراً فَلَمْ يَجِدُوا لَهُمْ مِنْ دُونِ اللّهِ أَنْصاراً

(آرى، سرانجام) همگى بخاطر گناهانشان غرق شدند و در آتش دوزخ وارد گشتند، و جز خدا ياورانى براى خود نيافتند.(25)

 

26 وَ قالَ نُوحٌ رَبِّ لا تَذَرْ عَلَى الأَرْضِ مِنَ الْكافِرِينَ دَيّاراً

نوح گفت: «پروردگارا! هيچ كس از كافران را بر روى زمين باقى مگذار!(26)

 

27 إِنَّكَ إِنْ تَذَرْهُمْ يُضِلُّوا عِبادَكَ وَ لا يَلِدُوا إِلاّ فاجِراً كَفّاراً

چرا كه اگر آنها را باقى بگذارى، بندگانت را گمراه مى كنند و جز نسلى فاجر و كافر به دنيا نمى آورند.(27)

 

28 رَبِّ اغْفِرْ لِي وَ لِوالِدَيَّ وَ لِمَنْ دَخَلَ بَيْتِيَ مُؤْمِناً وَ لِلْمُؤْمِنِينَ وَ الْمُؤْمِناتِ وَ لا تَزِدِ الظّالِمِينَ إِلاّ تَباراً

پروردگارا! مرا، و پدر و مادرم و همه كسانى را كه با ايمان وارد خانه من شدند، و جميع مردان و زنان با ايمان را بيامرز; و ستمكاران را جز هلاكت ميفزا!»(28)

[72] سوره جن

بِسمِ اللّهِ الرَّحمنِ الرَّحيمِ

به نام خداوند بخشنده مهربان

 

 

 

1 قُلْ أُوحِيَ إِلَيَّ أَنَّهُ اسْتَمَعَ نَفَرٌ مِنَ الْجِنِّ فَقالُوا إِنّا سَمِعْنا قُرْآناً عَجَباً

بگو: به من وحى شده است كه جمعى از جنّ به سخنانم گوش فرا داده اند، سپس گفته اند: «ما قرآن عجيبى شنيده ايم،(1)

 

2 يَهْدِي إِلَى الرُّشْدِ فَآمَنّا بِهِ وَ لَنْ نُشْرِكَ بِرَبِّنا أَحَداً

كه به راه راست هدايت مى كند، پس ما به آن ايمان آورده ايم و هرگز كسى را همتاى پروردگارمان قرار نمى دهيم،(2)

 

3 وَ أَنَّهُ تَعالى جَدُّ رَبِّنا مَا اتَّخَذَ صاحِبَةً وَ لا وَلَداً

و اين كه والا است مقام با عظمت پروردگار ما، و او هرگز براى خود همسر و فرزندى انتخاب نكرده است،(3)

 

4 وَ أَنَّهُ كانَ يَقُولُ سَفِيهُنا عَلَى اللّهِ شَطَطاً

و اين كه سفيه ما (ابليس) درباره خداوند سخنان ناروا مى گفت، (4)

 

5 وَ أَنّا ظَنَنّا أَنْ لَنْ تَقُولَ الإِنْسُ وَ الْجِنُّ عَلَى اللّهِ كَذِباً

و اين كه ما گمان مى كرديم كه انس و جنّ هرگز بر خدا دروغ نمى بندند،(5)

 

6 وَ أَنَّهُ كانَ رِجالٌ مِنَ الإِنْسِ يَعُوذُونَ بِرِجال مِنَ الْجِنِّ فَزادُوهُمْ رَهَقاً

و اين كه مردانى از بشر به مردانى از جنّ پناه مى بردند، و بر گمراهى و طغيان آنان مى افزودند،(6)

 

7 وَ أَنَّهُمْ ظَنُّوا كَما ظَنَنْتُمْ أَنْ لَنْ يَبْعَثَ اللّهُ أَحَداً

و اين كه آنها گمان كردند ـ همان گونه كه شما گمان مى كرديد كه خداوند هرگز كسى را (به نبوت) مبعوث نمى كند،(7)

 

8 وَ أَنّا لَمَسْنَا السَّماءَ فَوَجَدْناها مُلِئَتْ حَرَساً شَدِيداً وَ شُهُباً

و اين كه ما آسمان را جستجو كرديم و آن را پر از محافظان قوى و تيرهاى شهاب يافتيم،(8)

 

9 وَ أَنّا كُنّا نَقْعُدُ مِنْها مَقاعِدَ لِلسَّمْعِ فَمَنْ يَسْتَمِعِ الْآنَ يَجِدْ لَهُ شِهاباً رَصَداً

و اين كه ما پيش از اين به استراق سمع در آسمانها مى نشستيم; امّا اكنون هركس بخواهد استراق سمع كند، شهابى را در كمين خود مى يابد،(9)

 

10 وَ أَنّا لا نَدْرِي أَ شَرٌّ أُرِيدَ بِمَنْ فِي الأَرْضِ أَمْ أَرادَ بِهِمْ رَبُّهُمْ رَشَداً

و اين كه (با اين اوضاع) ما نمى دانيم آيا اراده شرّى درباره اهل زمين شده يا پروردگارشان خواسته است آنان را هدايت كند؟!(10)

 

11 وَ أَنّا مِنَّا الصّالِحُونَ وَ مِنّا دُونَ ذالِكَ كُنّا طَرائِقَ قِدَداً

و اين كه در ميان ما، افرادى صالح و افرادى غيرصالحند;و ما گروههاى متفاوتى هستيم،(11)

 

12 وَ أَنّا ظَنَنّا أَنْ لَنْ نُعْجِزَ اللّهَ فِي الأَرْضِ وَ لَنْ نُعْجِزَهُ هَرَباً

و اين كه ما يقين داريم هرگز نمى توانيم بر اراده خداوند در زمين غالب شويم و نمى توانيم از (پنجه قدرت) او بگريزيم،(12)

 

13 وَ أَنّا لَمّا سَمِعْنَا الْهُدى آمَنّا بِهِ فَمَنْ يُؤْمِنْ بِرَبِّهِ فَلا يَخافُ بَخْساً وَ لا رَهَقاً

و اين كه ما هنگامى كه هدايت (قرآن) را شنيديم به آن ايمان آورديم; و هركس به پروردگارش ايمان بياورد، نه از نقصان (در پاداش) مى ترسد و نه از ستم (در مجازات).(13)

 

14 وَ أَنّا مِنَّا الْمُسْلِمُونَ وَ مِنَّا الْقاسِطُونَ فَمَنْ أَسْلَمَ فَأُولئِكَ تَحَرَّوْا رَشَداً

و اين كه گروهى از ما مسلمان و گروهى ستمكارند; هركس اسلام را اختيار كند راه راست را برگزيده است،(14)

 

15 وَ أَمَّا الْقاسِطُونَ فَكانُوا لِجَهَنَّمَ حَطَباً

و امّا ستمكاران آتشگيره و هيزم دوزخند.(15)

 

16 وَ أ لَّوِ اسْتَقامُوا عَلَى الطَّرِيقَةِ لاَ َسْقَيْناهُمْ ماءً غَدَقاً

و اين كه اگر آنها [= جنّ و انس] در راه (ايمان) استقامت ورزند، با آب فراوان (باران) سيرابشان مى كنيم.(16)

 

17 لِنَفْتِنَهُمْ فِيهِ وَ مَنْ يُعْرِضْ عَنْ ذِكْرِ رَبِّهِ يَسْلُكْهُ عَذاباً صَعَداً

تا آنها را با اين نعمت فراوان بيازماييم; و هركس از ياد پروردگارش روى گرداند، او را به عذاب فزاينده اى گرفتار مى سازد!(17)

 

18 وَ أَنَّ الْمَساجِدَ لِلّهِ فَلا تَدْعُوا مَعَ اللّهِ أَحَداً

و اين كه مساجد از آنِ خداست، پس هيچ كس را با خدا نخوانيد.(18)

 

19 وَ أَنَّهُ لَمّا قامَ عَبْدُ اللّهِ يَدْعُوهُ كادُوا يَكُونُونَ عَلَيْهِ لِبَداً

و اين كه هنگامى كه بنده خدا [= محمد] به عبادت برمى خاست و او را مى خواند، گروهى پيرامون او بشدت ازدحام مى كردند.» (19)

 

20 قُلْ إِنَّما أَدْعُوا رَبِّي وَ لا أُشْرِكُ بِهِ أَحَداً

بگو: «من تنها پروردگارم را مى خوانم و هيچ كس را همتاى او قرار نمى دهم.»(20)

 

21 قُلْ إِنِّي لا أَمْلِكُ لَكُمْ ضَرّاً وَ لا رَشَداً

بگو: «من مالك زيان و هدايتى براى شما نيستم.» (21)

 

22 قُلْ إِنِّي لَنْ يُجِيرَنِي مِنَ اللّهِ أَحَدٌ وَ لَنْ أَجِدَ مِنْ دُونِهِ مُلْتَحَداً

بگو: «(اگر من نيز برخلاف فرمانش رفتار كنم) هيچ كس مرا در برابر خداوند حمايت نمى كند و پناهگاهى جز او نمى يابم;(22)

 

23 إِلاّ بَلاغاً مِنَ اللّهِ وَ رِسالاتِهِ وَ مَنْ يَعْصِ اللّهَ وَ رَسُولَهُ فَإِنَّ لَهُ نارَ جَهَنَّمَ خالِدِينَ فِيها أَبَداً

تنها وظيفه من ابلاغ ازسوى خدا و رساندن رسالات اوست; و كسانى كه نافرمانى خدا و پيامبرش كنند، آتش دوزخ از آن آنهاست و جاودانه در آن مى مانند.»(23)

 

24 حَتّى إِذا رَأَوْا ما يُوعَدُونَ فَسَيَعْلَمُونَ مَنْ أَضْعَفُ ناصِراً وَ أَقَلُّ عَدَداً

(اين كارشكنى كفّار همچنان ادامه مى يابد) تا هنگامى كه آنچه را به آنها وعده داده شده ببينند; آنگاه مى دانند چه كسى ياورش ضعيف تر و جمعيّتش كمتر است!(24)

 

25 قُلْ إِنْ أَدْرِي أَ قَرِيبٌ ما تُوعَدُونَ أَمْ يَجْعَلُ لَهُ رَبِّي أَمَداً

بگو: «من نمى دانم آنچه به شما وعده داده شده نزديك است يا پروردگارم زمانى براى آن قرار مى دهد؟!(25)

 

26 عالِمُ الْغَيْبِ فَلا يُظْهِرُ عَلى غَيْبِهِ أَحَداً

داناى غيب اوست و هيچ كس را بر اسرار غيبش آگاه نمى سازد، (26)

 

27 إِلاّ مَنِ ارْتَضى مِنْ رَسُول فَإِنَّهُ يَسْلُكُ مِنْ بَيْنِ يَدَيْهِ وَ مِنْ خَلْفِهِ رَصَداً

مگر پيامبرانى كه (آنان را برگزيده و) از آنها راضى است و مراقبينى از پيش رو و پشت سر براى آنها قرار مى دهد،(27)

 

28 لِيَعْلَمَ أَنْ قَدْ أَبْلَغُوا رِسالاتِ رَبِّهِمْ وَ أَحاطَ بِما لَدَيْهِمْ وَ أَحْصى كُلَّ شَيْء عَدَداً

تا معلوم شود پيامبرانش رسالتهاى پروردگارشان را ابلاغ كرده اند; و او به آنچه نزد آنهاست احاطه دارد و همه چيز را احصا كرده است.»(28)

[73] سوره مزمل

بِسمِ اللّهِ الرَّحمنِ الرَّحيمِ

به نام خداوند بخشنده مهربان

 

 

 

1 يا أَيُّهَا الْمُزَّمِّلُ

اى جامه به خود پيچيده!(1)

 

2 قُمِ اللَّيْلَ إِلاّ قَلِيلاً

شب را، جز كمى، به پاخيز.(2)

 

3 نِصْفَهُ أَوِ انْقُصْ مِنْهُ قَلِيلاً

نيمى از شب را، يا كمى از آن كم كن،(3)

 

4 أَوْ زِدْ عَلَيْهِ وَ رَتِّلِ الْقُرْآنَ تَرْتِيلاً

يا بر نصف آن بيفزا، وقرآن را با تامّل و دقت بخوان.(4)

 

5 إنّا سَنُلْقى عَلَيْكَ قَوْلاً ثَقيلا

چرا كه ما بزودى سخنى سنگين به تو القا خواهيم كرد.(5)

 

6 إِنَّ ناشِئَةَ اللَّيْلِ هِيَ أَشَدُّ وَطْئاً وَ أَقْوَمُ قِيلاً

به يقين نماز و عبادت شبانه گامى استوارتر و گفتارى پايدارتر است.(6)

 

7 إِنَّ لَكَ فِي النَّهارِ سَبْحاً طَوِيلاً

زيرا تو در روز تلاش مستمر و طولانى خواهى داشت (و عبادت شبانه به تو نيرو مى بخشد).(7)

 

8 وَ اذْكُرِ اسْمَ رَبِّكَ وَ تَبَتَّلْ إِلَيْهِ تَبْتِيلاً

نام پروردگارت را ياد كن و تنها به او دل ببند.(8)

 

9 رَبُّ الْمَشْرِقِ وَ الْمَغْرِبِ لا إِلهَ إِلاّ هُوَ فَاتَّخِذْهُ وَكِيلاً

اوست پروردگار مشرق و مغرب كه معبودى جز او نيست، پس او را تكيه گاه خود انتخاب كن،(9)

 

10 وَ اصْبِرْ عَلى ما يَقُولُونَ وَ اهْجُرْهُمْ هَجْراً جَمِيلاً

و در برابر آنچه (دشمنان) مى گويند شكيبا باش و به طرزى شايسته از آنان دورى گزين.(10)

 

11 وَ ذَرْنِي وَ الْمُكَذِّبِينَ أُولِي النَّعْمَةِ وَ مَهِّلْهُمْ قَلِيلاً

و مرا با تكذيب كنندگان صاحب نعمت واگذار، و آنها را كمى مهلت ده،(11)

 

12 إِنَّ لَدَيْنا أَنْكالاً وَ جَحِيماً

كه (براى آنها) نزد ما غل و زنجيرها و (آتش) دوزخ است،(12)

 

13 وَ طَعاماً ذا غُصَّة وَ عَذاباً أَلِيماً

و غذايى گلوگير، و عذابى دردناك،(13)

 

14 يَوْمَ تَرْجُفُ الأَرْضُ وَ الْجِبالُ وَ كانَتِ الْجِبالُ كَثِيباً مَهِيلاً

در آن روز كه زمين و كوهها سخت به لرزه درمى آيد، و كوهها (چنان درهم كوبيده مى شود كه) به شكل توده هايى از شن نرم در مى آيد.(14)

 

15 إِنّا أَرْسَلْنا إِلَيْكُمْ رَسُولاً شاهِداً عَلَيْكُمْ كَما أَرْسَلْنا إِلى فِرْعَوْنَ رَسُولاً

ما پيامبرى به سوى شما فرستاديم كه گواه بر شماست، همان گونه كه به سوى فرعون پيامبرى فرستاديم.(15)

 

16 فَعَصى فِرْعَوْنُ الرَّسُولَ فَأَخَذْناهُ أَخْذاً وَبِيلاً

(ولى) فرعون به مخالفت و نافرمانى آن پيامبر برخاست، و ما او را سخت مجازات كرديم.(16)

 

17 فَكَيْفَ تَتَّقُونَ إِنْ كَفَرْتُمْ يَوْماً يَجْعَلُ الْوِلْدانَ شِيباً

شما (نيز) اگر كافر شويد، چگونه خود را (از عذاب) روزى كه كودكان را پير مى كند بركنار مى داريد؟!(17)

 

18 السَّماءُ مُنْفَطِرٌ بِهِ كانَ وَعْدُهُ مَفْعُولاً

و در آن روز آسمان از هم شكافته مى شود، و وعده الهى، انجام شدنى است.(18)

 

19 إِنَّ هذِهِ تَذْكِرَةٌ فَمَنْ شاءَ اتَّخَذَ إِلى رَبِّهِ سَبِيلاً

اين هشدار و تذكّرى است، پس هركس بخواهد راهى به سوى پروردگارش برمى گزيند.(19)

 

20 إِنَّ رَبَّكَ يَعْلَمُ أَنَّكَ تَقُومُ أَدْنى مِنْ ثُلُثَيِ اللَّيْلِ وَ نِصْفَهُ وَ ثُلُثَهُ وَ طائِفَةٌ مِنَ الَّذِينَ مَعَكَ وَ اللّهُ يُقَدِّرُ اللَّيْلَ وَ النَّهارَ عَلِمَ أَنْ لَنْ تُحْصُوهُ فَتابَ عَلَيْكُمْ فَاقْرَؤُا ما تَيَسَّرَ مِنَ الْقُرْآنِ عَلِمَ أَنْ سَيَكُونُ مِنْكُمْ مَرْضى وَ آخَرُونَ يَضْرِبُونَ فِي الأَرْضِ يَبْتَغُونَ مِنْ فَضْلِ اللّهِ وَ آخَرُونَ يُقاتِلُونَ فِي سَبِيلِ اللّهِ فَاقْرَؤُا ما تَيَسَّرَ مِنْهُ وَ أَقِيمُوا الصَّلاةَ وَ آتُوا الزَّكاةَ وَ أَقْرِضُوا اللّهَ قَرْضاً حَسَناً وَ ما تُقَدِّمُوا لاِ َنْفُسِكُمْ مِنْ خَيْر تَجِدُوهُ عِنْدَ اللّهِ هُوَ خَيْراً وَ أَعْظَمَ أَجْراً وَ اسْتَغْفِرُوا اللّهَ إِنَّ اللّهَ غَفُورٌ رَحِيمٌ

پروردگارت مى داند كه تو و گروهى از آنها كه با تو هستند نزديك دو ثلث از شب يا نصف يا ثلث آن را به پا مى خيزيد; (و به عبادت و تلاوت قرآن مى پردازيد) خداوند شب و روز را اندازه گيرى مى كند; او مى داند كه شما نمى توانيد مقدار آن را (به دقت براى عبادت كردن) اندازه گيرى كنيد، پس شما را بخشيد; اكنون آنچه براى شما ميسّر است قرآن بخوانيد. او مى داند بزودى گروهى از شما بيمار مى شوند، و گروهى ديگر براى به دست آوردن فضل الهى (و كسب روزى) به سفر مى روند، و گروهى ديگر در راه خدا جهاد مى كنند (و از تلاوت قرآن باز مى مانند)، پس آنچه براى شما ممكن است از آن تلاوت كنيد و نماز را برپا داريد و زكات بپردازيد و به خدا «قرض الحسنه» دهيد (در راه او انفاق نماييد) و (بدانيد) آنچه از كارهاى نيك براى خود از پيش مى فرستيد آن را نزد خدا به بهترين وجه و بزرگترين پاداش خواهيد يافت; و از خدا آمرزش بطلبيد كه خداوند آمرزنده و مهربان است. (20)

[74] سوره مدثر

بِسمِ اللّهِ الرَّحمنِ الرَّحيمِ

به نام خداوند بخشنده مهربان

 

 

 

1 يا أَيُّهَا الْمُدَّثِّرُ

اى جامه خواب به خود پيچيده (و در بستر آرميده)!(1)

 

2 قُمْ فَأَنْذِرْ

برخيز و انذار كن، (و جهانيان را بيم ده)(2)

 

3 وَ رَبَّكَ فَكَبِّرْ

و پروردگارت را بزرگ بشمار(3)

 

4 وَ ثِيابَكَ فَطَهِّرْ

و لباست را پاكيزه دار،(4)

 

5 وَ الرُّجْزَ فَاهْجُرْ

و از پليدى دورى كن،(5)

 

6 وَ لا تَمْنُنْ تَسْتَكْثِرُ

و منّت مگذار و عملت را بزرگ مشمار،(6)

 

7 وَ لِرَبِّكَ فَاصْبِرْ

و بخاطر پروردگارت شكيبايى كن.(7)

 

8 فَإِذا نُقِرَ فِي النّاقُورِ

و (بدانيد) هنگامى كه در «صور» دميده شود،(8)

 

9 فَذالِكَ يَوْمَئِذ يَوْمٌ عَسِيرٌ

آن روز، روز سختى است،(9)

 

10 عَلَى الْكافِرِينَ غَيْرُ يَسِير

كه براى كافران آسان نيست!(10)

 

11 ذَرْنِي وَ مَنْ خَلَقْتُ وَحِيداً

مرا با كسى كه او را به تنهايى آفريده ام واگذار! (11)

 

12 وَ جَعَلْتُ لَهُ مالاً مَمْدُوداً

همان كس كه براى او مال گسترده اى قرار دادم،(12)

 

13 وَ بَنِينَ شُهُوداً

و فرزندان فراوانى كه همواره در كنار او (و در خدمت او) هستند،(13)

 

14 وَ مَهَّدْتُ لَهُ تَمْهِيداً

و وسايل زندگى را (از هر نظر) برايش فراهم ساختم!(14)

 

15 ثُمَّ يَطْمَعُ أَنْ أَزِيدَ

باز طمع دارد كه (به آن) بيفزايم.(15)

 

16 كَلاّ إِنَّهُ كانَ لاِياتِنا عَنِيداً

هرگز. چنين نخواهد شد; چرا كه او نسبت به آيات ما دشمنى مىورزد.(16)

 

17 سَأُرْهِقُهُ صَعُوداً

و بزودى او را مجبور مى كنم كه از قله زندگى بالا رود. (پس او را به زير مى افكنم)!(17)

 

18 إِنَّهُ فَكَّرَ وَ قَدَّرَ

او (براى مبارزه با قرآن) انديشه كرد و نقشه اى (شيطانى) كشيد!(18)

 

19 فَقُتِلَ كَيْفَ قَدَّرَ

مرگ بر او! چگونه نقشه اى كشيد؟!(19)

 

20 ثُمَّ قُتِلَ كَيْفَ قَدَّرَ

باز هم مرگ بر او، چگونه نقشه كشيد؟!(20)

 

21 ثُمَّ نَظَرَ

پس نگاهى كرد،(21)

 

22 ثُمَّ عَبَسَ وَ بَسَرَ

و چهره درهم كشيد و عجولانه دست به كار شد;(22)

 

23 ثُمَّ أَدْبَرَ وَ اسْتَكْبَرَ

سپس پشت كرد و تكبّر ورزيد،(23)

 

24 فَقالَ إِنْ هذا إِلاّ سِحْرٌ يُؤْثَرُ

و سرانجام گفت: «اين (قرآن) چيزى جز سحر پيشينيان نيست!(24)

 

25 إِنْ هذا إِلاّ قَوْلُ الْبَشَرِ

اين فقط سخن انسان است.»(25)

 

26 سَأُصْلِيهِ سَقَرَ

(امّا) بزودى او را وارد سَقَر [=دوزخ] مى كنم!(26)

 

27 وَ ما أَدْراكَ ما سَقَرُ

و توچه مى دانى «سَقَر» چيست!(27)

 

28 لا تُبْقِي وَ لا تَذَرُ

(آتشى است كه) نه چيزى را باقى مى گذارد و نه رها مى سازد.(28)

 

29 لَوّاحَةٌ لِلْبَشَرِ

پوست را دگرگون مى كند.(29)

 

30 عَلَيْها تِسْعَةَ عَشَرَ

نوزده (نگهبان) بر آن گمارده شده اند!(30)

 

31 وَ ما جَعَلْنا أَصْحابَ النّارِ إِلاّ مَلائِكَةً وَ ما جَعَلْنا عِدَّتَهُمْ إِلاّ فِتْنَةً لِلَّذِينَ كَفَرُوا لِيَسْتَيْقِنَ الَّذِينَ أُوتُوا الْكِتابَ وَ يَزْدادَ الَّذِينَ آمَنُوا إِيماناً وَ لا يَرْتابَ الَّذِينَ أُوتُوا الْكِتابَ وَ الْمُؤْمِنُونَ وَ لِيَقُولَ الَّذِينَ فِي قُلُوبِهِمْ مَرَضٌ وَ الْكافِرُونَ ما ذا أَرادَ اللّهُ بِهذا مَثَلاً كَذالِكَ يُضِلُّ اللّهُ مَنْ يَشاءُ وَ يَهْدِي مَنْ يَشاءُ وَ ما يَعْلَمُ جُنُودَ رَبِّكَ إِلاّ هُوَ وَ ما هِيَ إِلاّ ذِكْرى لِلْبَشَرِ

نگهبانان دوزخ را فقط فرشتگان (عذاب) قرار داديم، و تعداد آنها را جز براى آزمايش كافران معين نكرديم تااهل كتاب [= يهود و نصارى] يقين پيدا كنند و بر ايمان مؤمنان بيفزايد، و اهل كتاب و مؤمنان (در حقّانيت اين كتاب آسمانى) ترديد به خود راه ندهند، و بيماردلان و كافران بگويند: «خدا از اين سخن چه منظورى دارد؟!» (آرى) اين گونه خداوند هركس را بخواهد گمراه مى سازد و هركس را بخواهد هدايت مى كند. و لشكريان پروردگارت را جز او كسى نمى داند، و اين جز هشدار و تذكرى براى انسانها نيست!(31)

 

32 كَلاّ وَ الْقَمَرِ

هرگز چنين نيست سوگند به ماه،(32)

 

33 وَ اللَّيْلِ إِذْ أَدْبَرَ

و به شب، هنگامى كه (دامن برچينيد و) پشت كند،(33)

 

34 وَ الصُّبْحِ إِذا أَسْفَرَ

و به صبح هنگامى كه چهره بگشايد،(34)

 

35 إِنَّها لاَِحْدَى الْكُبَرِ

كه آن (حوادث هولناك قيامت) از مسائل مهم است!(35)

 

36 نَذِيراً لِلْبَشَرِ

هشدارى است براى همه انسانها،(36)

 

37 لِمَنْ شاءَ مِنْكُمْ أَنْ يَتَقَدَّمَ أَوْ يَتَأَخَّرَ

براى كسانى از شما كه مى خواهند (به سوى هدايت) پيش روند يا عقب بمانند(37)

 

38 كُلُّ نَفْس بِما كَسَبَتْ رَهِينَةٌ

(آرى) هركس در گرو اعمال خويش است،(38)

 

39 إِلاّ أَصْحابَ الْيَمِينِ

مگر «اصحاب يمين» (كه نامه اعمالشان را به دست راستشان مى دهند)!(39)

 

40 فِي جَنّات يَتَساءَلُونَ

آنها در باغهاى بهشتى اند، و سؤال مى كنند،(40)

 

41 عَنِ الْمُجْرِمِينَ

از مجرمان:(41)

 

42 ما سَلَكَكُمْ فِي سَقَرَ

«چه چيز شما را به دوزخ وارد ساخت؟!»(42)

 

43 قالُوا لَمْ نَكُ مِنَ الْمُصَلِّينَ

مى گويند: «ما از نمازگزاران نبوديم،(43)

 

44 وَ لَمْ نَكُ نُطْعِمُ الْمِسْكِينَ

و اطعام مستمند نمى كرديم،(44)

 

45 وَ كُنّا نَخُوضُ مَعَ الْخائِضِينَ

و پيوسته با اهل باطل همنشين و همصدا بوديم،(45)

 

46 وَ كُنّا نُكَذِّبُ بِيَوْمِ الدِّينِ

و همواره روز جزا را انكار مى كرديم،(46)

 

47 حَتّى أَتانَا الْيَقِينُ

تا زمانى كه مرگ ما فرا رسيد!»(47)

 

48 فَما تَنْفَعُهُمْ شَفاعَةُ الشّافِعِينَ

از اين رو شفاعت شفاعت كنندگان به حال آنها سودى نمى بخشد.(48)

 

49 فَما لَهُمْ عَنِ التَّذْكِرَةِ مُعْرِضِينَ

چرا آنها از تذكّر روى گردانند؟!(49)

 

50 كَأَنَّهُمْ حُمُرٌ مُسْتَنْفِرَةٌ

گويى گورخرانى رميده اند،(50)

 

51 فَرَّتْ مِنْ قَسْوَرَة

كه از (مقابل) شيرى فرار كرده اند!(51)

 

52 بَلْ يُرِيدُ كُلُّ امْرِئ مِنْهُمْ أَنْ يُؤْتى صُحُفاً مُنَشَّرَةً

بلكه هر كدام از آنها انتظار دارد نامه جداگانه اى (از سوى خدا) به او داده شود.(52)

 

53 كَلاّ بَلْ لا يَخافُونَ الآخِرَةَ

هرگز چنين نيست (كه آنان مى گويند،)بلكه آنها از آخرت نمى ترسند.(53)

 

54 كَلاّ إِنَّهُ تَذْكِرَةٌ

 هرگز چنين نيست (كه آنها مى گويند) آن (قرآن) يك تذكر و يادآورى است.(54)

 

55 فَمَنْ شاءَ ذَكَرَهُ

و كسانى كه بخواهند از آن پند مى گيرند;(55)

 

56 وَ ما يَذْكُرُونَ إِلاّ أَنْ يَشاءَ اللّهُ هُوَ أَهْلُ التَّقْوى وَ أَهْلُ الْمَغْفِرَةِ

و آنها پند نمى گيرند مگر اين كه خدا بخواهد; او اهل تقوا و اهل آمرزش است (و به هر كس آنچه لايق باشد مى دهد).(56)

[75] سوره قيامه

بِسمِ اللّهِ الرَّحمنِ الرَّحيمِ

به نام خداوند بخشنده مهربان

 

 

 

1 لا أُقْسِمُ بِيَوْمِ الْقِيامَةِ

سوگند به روز قيامت،(1)

 

2 وَ لا أُقْسِمُ بِالنَّفْسِ اللَّوّامَةِ

و سوگند به نفس ملامتگر (و وجدان بيدار كه رستاخيز حق است)!(2)

 

3 أَ يَحْسَبُ الإِنْسانُ أَلَّنْ نَجْمَعَ عِظامَهُ

آيا انسان مى پندارد كه هرگز استخوانهاى او را جمع نخواهيم كرد؟! (3)

 

4 بَلى قادِرِينَ عَلى أَنْ نُسَوِّيَ بَنانَهُ

آرى قادريم كه (حتى خطوط سر) انگشتان او را موزون و مرتب كنيم.(4)

 

5 بَلْ يُرِيدُ الإِنْسانُ لِيَفْجُرَ أَمامَهُ

(انسان شك در معاد ندارد) بلكه او مى خواهد (آزادانه) در تمام عمر گناه كند.(5)

 

6 يَسْئَلُ أَيّانَ يَوْمُ الْقِيامَةِ

(از اين رو) مى پرسد: «روز قيامت كى خواهد بود»؟!(6)

 

7 فَإِذا بَرِقَ الْبَصَرُ

(بگو:) در آن هنگام كه چشمها از شدّت وحشت خيره گردد، (7)

 

8 وَ خَسَفَ الْقَمَرُ

و ماه بى نور شود،(8)

 

9 وَ جُمِعَ الشَّمْسُ وَ الْقَمَرُ

و خورشيد و ماه با هم جمع شوند،(9)

 

10 يَقُولُ الإِنْسانُ يَوْمَئِذ أَيْنَ الْمَفَرُّ

آن روز انسان مى گويد: «راه فرار كجاست؟!»(10)

 

11 كَلاّ لا وَزَرَ

هرگز چنين نيست، (راه فرار و) پناهگاهى وجود ندارد!(11)

 

12 إِلى رَبِّكَ يَوْمَئِذ الْمُسْتَقَرُّ

آن روز قرارگاه نهايى تنها به سوى پروردگار تو است;(12)

 

13 يُنَبَّؤُا الإِنْسانُ يَوْمَئِذ بِما قَدَّمَ وَ أَخَّرَ

و در آن روز انسان را از آنچه كه از پيش و پشت سر فرستاده آگاه مى كنند.(13)

 

14 بَلِ الإِنْسانُ عَلى نَفْسِهِ بَصِيرَةٌ

بلكه انسان از وضع خويش آگاه است،(14)

 

15 وَ لَوْ أَلْقى مَعاذِيرَهُ

هرچند (در ظاهر) براى خود عذرهايى بتراشد. (15)

 

16 لا تُحَرِّكْ بِهِ لِسانَكَ لِتَعْجَلَ بِهِ

(هنگام نزول وحى) زبانت را بخاطر عجله براى خواندن آن حركت مده، (16)

 

17 إِنَّ عَلَيْنا جَمْعَهُ وَ قُرْآنَهُ

چرا كه جمع كردن و خواندن آن بر عهده ماست.(17)

 

18 فَإِذا قَرَأْناهُ فَاتَّبِعْ قُرْآنَهُ

پس هر گاه آن را خوانديم، از خواندن آن پيروى كن.(18)

 

19 ثُمَّ إِنَّ عَلَيْنا بَيانَهُ

سپس بيان (و توضيح) آن (نيز) بر عهده ماست.(19)

 

20 كَلاّ بَلْ تُحِبُّونَ الْعاجِلَةَ

هرگز چنين نيست (كه شما مى پنداريد!) بلكه شما دنياى زودگذر را دوست داريد(20)

 

21 وَ تَذَرُونَ الآخِرَةَ

وآخرت را رها مى كنيد.(21)

 

22 وُجُوهٌ يَوْمَئِذ ناضِرَةٌ

(آرى) در آن روز صورتهايى شاداب و مسرور است،(22)

 

23 إِلى رَبِّها ناظِرَةٌ

و به (الطاف) پروردگارشان مى نگرند.(23)

 

24 وَ وُجُوهٌ يَوْمَئِذ باسِرَةٌ

و در آن روز صورتهايى عبوس و درهم كشيده است،(24)

 

25 تَظُنُّ أَنْ يُفْعَلَ بِها فاقِرَةٌ

(زيرا) مى دانند عذابى كمرشكن در پيش دارند!(25)

 

26 كَلاّ إِذا بَلَغَتِ التَّراقِيَ

هرگز چنين نيست (او ايمان نمى آورد) تا موقعى كه جان به گلوگاهش رسد،(26)

 

27 وَ قِيلَ مَنْ راق

و گفته شود: «آيا كسى هست كه (اين بيمار را از مرگ) نجات دهد؟!»(27)

 

28 وَ ظَنَّ أَنَّهُ الْفِراقُ

(در اين هنگام) او به جدايى از دنيا يقين پيدا مى كند،(28)

 

29 وَ الْتَفَّتِ السّاقُ بِالسّاقِ

و ساق پاها (از سختى جان دادن) به هم مى پيچد!(29)

 

30 إِلى رَبِّكَ يَوْمَئِذ الْمَساقُ

در آن روز مسير همه به سوى (دادگاه) پروردگارت خواهد بود.(30)

 

31 فَلا صَدَّقَ وَ لا صَلّى

(در آن روز گفته مى شود:) او ايمان نياورد و نماز نخواند،(31)

 

32 وَ لكِنْ كَذَّبَ وَ تَوَلّى

بلكه تكذيب كرد و روى گردان شد،(32)

 

33 ثُمَّ ذَهَبَ إِلى أَهْلِهِ يَتَمَطّى

سپس به سوى خانواده خود بازگشت در حالى كه متكبرانه قدم بر مى داشت.(33)

 

34 أَوْلى لَكَ فَأَوْلى

(با اين اعمال) عذاب الهى براى تو شايسته تر است، شايسته تر! (34)

 

35 ثُمَّ أَوْلى لَكَ فَأَوْلى

باز هم عذاب براى تو شايسته تر است، شايسته تر!(35)

 

36 أَ يَحْسَبُ الإِنْسانُ أَنْ يُتْرَكَ سُدىً

آيا انسان گمان مى كند بى هدف رها مى شود؟!(36)

 

37 أَ لَمْ يَكُ نُطْفَةً مِنْ مَنِيّ يُمْنى

آيا او نطفه اى از منى كه در رحم ريخته مى شود نبود؟! (37)

 

38 ثُمَّ كانَ عَلَقَةً فَخَلَقَ فَسَوّى

سپس بصورت خون بسته درآمد، و خداوند او را آفريد و موزون ساخت،(38)

 

39 فَجَعَلَ مِنْهُ الزَّوْجَيْنِ الذَّكَرَ وَ الأُنْثى

و از آن، دو زوج مرد و زن آفريد.(39)

 

40 أَ لَيْسَ ذالِكَ بِقادِر عَلى أَنْ يُحْيِيَ الْمَوْتى

آيا چنين كسى قادر نيست كه مردگان را زنده كند؟!(40)

[76] سوره انسان

بِسمِ اللّهِ الرَّحمنِ الرَّحيمِ

به نام خداوند بخشنده مهربان

 

 

 

1 هَلْ أَتى عَلَى الإِنْسانِ حِينٌ مِنَ الدَّهْرِ لَمْ يَكُنْ شَيْئاً مَذْكُوراً

آيا اين گونه نيست كه زمانى طولانى از روزگار بر انسان گذشت كه چيز قابل ذكرى نبود؟!(1)

 

2 إِنّا خَلَقْنَا الإِنْسانَ مِنْ نُطْفَة أَمْشاج نَبْتَلِيهِ فَجَعَلْناهُ سَمِيعاً بَصِيراً

ما انسان را از نطفه به هم آميخته اى آفريديم، و او را مى آزماييم; (بدين جهت) او را شنوا و بينا قرار داديم.(2)

 

3 إِنّا هَدَيْناهُ السَّبِيلَ إِمّا شاكِراً وَ إِمّا كَفُوراً

ما راه را به او نشان داديم، خواه شاكر باشد (و پذيرا گردد) يا ناسپاس.(3)

 

4 إِنّا أَعْتَدْنا لِلْكافِرِينَ سَلاسِلاْ وَ أَغْلالاً وَ سَعِيراً

ما براى كافران،زنجيرها و غلها و شعله هاى سوزان آتش آماده كرده ايم!(4)

 

5 إِنَّ الأَبْرارَ يَشْرَبُونَ مِنْ كَأْس كانَ مِزاجُها كافُوراً

به يقين ابرار (و نيكان) از جامى مى نوشند كه با عطر خوشى آميخته است،(5)

 

6 عَيْناً يَشْرَبُ بِها عِبادُ اللّهِ يُفَجِّرُونَها تَفْجِيراً

از چشمه اى كه بندگان خاصّ خدا از آن مى نوشند، و (از هر جا بخواهند) آن را جارى مى سازند.(6)

 

7 يُوفُونَ بِالنَّذْرِ وَ يَخافُونَ يَوْماً كانَ شَرُّهُ مُسْتَطِيراً

آنها به نذر خود وفا مى كنند، و از روزى كه شرّ و عذابش گسترده است مى ترسند،(7)

 

8 وَ يُطْعِمُونَ الطَّعامَ عَلى حُبِّهِ مِسْكِيناً وَ يَتِيماً وَ أَسِيراً

و غذاى (خود) را با اين كه به آن علاقه (و نياز) دارند، به «مستمند» و «يتيم» و «اسير» اطعام مى كنند.(8)

 

9 إِنَّما نُطْعِمُكُمْ لِوَجْهِ اللّهِ لا نُرِيدُ مِنْكُمْ جَزاءً وَ لا شُكُوراً

(و مى گويند:) ما شما را بخاطر خدا اطعام مى كنيم، وهيچ پاداش و سپاسى از شما نمى خواهيم.(9)

 

10 إِنّا نَخافُ مِنْ رَبِّنا يَوْماً عَبُوساً قَمْطَرِيراً

ما از پروردگارمان خائفيم در آن روزى كه عبوس و سخت است.(10)

 

11 فَوَقاهُمُ اللّهُ شَرَّ ذالِكَ الْيَوْمِ وَ لَقّاهُمْ نَضْرَةً وَ سُرُوراً

(بخاطر اين عقيده و عمل)خداوند آنان را از شرّ آن روز نگه مى دارد و از آنها استقبال مى كند درحالى كه شادمان و مسرورند.(11)

 

12 وَ جَزاهُمْ بِما صَبَرُوا جَنَّةً وَ حَرِيراً

و در برابر صبرشان، بهشت و لباسهاى حرير (بهشتى) را به آنها پاداش مى دهد. (12)

 

13 مُتَّكِئِينَ فِيها عَلَى الأَرائِكِ لا يَرَوْنَ فِيها شَمْساً وَ لا زَمْهَرِيراً

در آنجا بر تختهاى زيبا تكيه كرده اند، نه آفتاب (سوزان) را در آن جا مى بينند و نه سرما را.(13)

 

14 وَ دانِيَةً عَلَيْهِمْ ظِلالُها وَ ذُلِّلَتْ قُطُوفُها تَذْلِيلاً

و در حالى است كه سايه هاى آن (درختان بهشتى) بر آنها فرو افتاده و چيدن ميوه هايش بسيار آسان است.(14)

 

15 وَ يُطافُ عَلَيْهِمْ بِآنِيَة مِنْ فِضَّة وَ أَكْواب كانَتْ قَوارِيرَا

و در گردا گرد آنها ظرفهايى سيمين و قدحهايى بلورين مى گردانند، (پر از بهترين غذاها و نوشيدنيها)(15)

 

16 قَوارِيرَا مِنْ فِضَّة قَدَّرُوها تَقْدِيراً

ظرفهاى بلورينى از نقره (شفاف)، كه آنها را به اندازه مناسب آماده كرده اند.(16)

 

17 وَ يُسْقَوْنَ فِيها كَأْساً كانَ مِزاجُها زَنْجَبِيلاً

و در آن جا از جامهايى سيراب مى شوند لبريز (از شراب طهورى) كه آميخته با زنجبيل است، (17)

 

18 عَيْناً فِيها تُسَمّى سَلْسَبِيلاً

از چشمه اى در آنجا كه نامش سلسبيل است.(18)

 

19 وَ يَطُوفُ عَلَيْهِمْ وِلْدانٌ مُخَلَّدُونَ إِذا رَأَيْتَهُمْ حَسِبْتَهُمْ لُؤْلُؤاً مَنْثُوراً

و بر گردشان (براى پذيرايى) نوجوانانى جاودانى مى گردند كه هرگاه آنها را ببينى گمان مى كنى مرواريد پراكنده اند.(19)

 

20 وَ إِذا رَأَيْتَ ثَمَّ رَأَيْتَ نَعِيماً وَ مُلْكاً كَبِيراً

و هنگامى كه آن جا را ببينى نعمت و ملك عظيمى را مى بينى!(20)

 

21 عالِيَهُمْ ثِيابُ سُندُس خُضْرٌ وَ إِسْتَبْرَقٌ وَ حُلُّوا أَساوِرَ مِنْ فِضَّة وَ سَقاهُمْ رَبُّهُمْ شَراباً طَهُوراً

بر اندام آنها [= بهشتيان] لباسهايى است از حرير نازك سبز رنگ، و از ديباى ضخيم، و با دستبندهايى از نقره آراسته شده اند، و پروردگارشان شراب طهور به آنان مى نوشاند.(21)

 

22 إِنَّ هذا كانَ لَكُمْ جَزاءً وَ كانَ سَعْيُكُمْ مَشْكُوراً

اين پاداش شماست، و سعى و تلاش شما مورد قدردانى است.(22)

 

23 إِنّا نَحْنُ نَزَّلْنا عَلَيْكَ الْقُرْآنَ تَنْزِيلاً

به يقين قرآن را ما بر تو نازل كرديم.(23)

 

24 فَاصْبِرْ لِحُكْمِ رَبِّكَ وَ لا تُطِعْ مِنْهُمْ آثِماً أَوْ كَفُوراً

پس در (تبليغ و اجراى) حكم پروردگارت شكيبا و با استقامت باش، و از هيچ گنهكار يا كافرى از آنان اطاعت مكن.(24)

 

25 وَ اذْكُرِ اسْمَ رَبِّكَ بُكْرَةً وَ أَصِيلاً

و نام پروردگارت را هر صبح و شام ياد كن.(25)

 

26 وَ مِنَ اللَّيْلِ فَاسْجُدْ لَهُ وَ سَبِّحْهُ لَيْلاً طَوِيلاً

و پاسى از شب را براى او سجده كن، و مقدارى طولانى از شب، او را تسبيح گوى. (26)

 

27 إِنَّ هؤُلاءِ يُحِبُّونَ الْعاجِلَةَ وَ يَذَرُونَ وَراءَهُمْ يَوْماً ثَقِيلاً

آنها زندگى زودگذر دنيا را دوست دارند، در حالى كه روز سختى را پشت سر خودرها مى كنند.(27)

 

28 نَحْنُ خَلَقْناهُمْ وَ شَدَدْنا أَسْرَهُمْ وَ إِذا شِئْنا بَدَّلْنا أَمْثالَهُمْ تَبْدِيلاً

ما آنها را آفريديم و پيوندهاى وجودشان را محكم كرديم، و هر زمان بخواهيم جاى آنان را به گروهى همانند آنان مى دهيم.(28)

 

29 إِنَّ هذِهِ تَذْكِرَةٌ فَمَنْ شاءَ اتَّخَذَ إِلى رَبِّهِ سَبِيلاً

اين يك تذكّر و يادآورى است، و هركس بخواهد (با استفاده از آن) راهى به سوى پروردگارش برمى گزيند.(29)

 

30 وَ ما تَشاؤُنَ إِلاّ أَنْ يَشاءَ اللّهُ إِنَّ اللّهَ كانَ عَلِيماً حَكِيماً

و شما (چيزى را) نمى خواهيد مگر اين كه خدا بخواهد،خداوند دانا و حكيم بوده و هست.(30)

 

31يُدْخِلُ مَنْ يَشاءُ فِي رَحْمَتِهِ وَ الظّالِمِينَ أَعَدَّ لَهُمْ عَذاباً أَلِيماً

هركس را بخواهد (و شايسته باشد) در رحمت (وسيع) خود وارد مى كند، و براى ستمكاران عذاب دردناكى آماده ساخته است!(31)

[77] سوره مرسلات

بِسمِ اللّهِ الرَّحمنِ الرَّحيمِ

به نام خداوند بخشنده مهربان

 

 

 

1 وَ الْمُرْسَلاتِ عُرْفاً

سوگند به فرشتگانى كه پى در پى فرستاده مى شوند،(1)

 

2 فَالْعاصِفاتِ عَصْفاً

و آنها كه همچون تندباد حركت مى كنند،(2)

 

3 وَ النّاشِراتِ نَشْراً

و سوگند به آنها كه (ابرها را) مى گسترانند،(3)

 

4 فَالْفارِقاتِ فَرْقاً

و آنها كه جدا مى كنند،(4)

 

5 فَالْمُلْقِياتِ ذِكْراً

و سوگند به آنها كه آيات بيدارگر (الهى) را (به انبيا) القا مى نمايند،(5)

 

6 عُذْراً أَوْ نُذْراً

براى اتمام حجّت يا انذار، (6)

 

7 إِنَّما تُوعَدُونَ لَواقِعٌ

(سوگند به همه اين امور) كه آنچه به شما (درباره قيامت) وعده داده مى شود، واقع شدنى است!(7)

 

8 فَإِذَا النُّجُومُ طُمِسَتْ

در آن هنگام كه ستارگان محو و تاريك شوند (8)

 

9 وَ إِذَا السَّماءُ فُرِجَتْ

و (كرات) آسمان از هم بشكافند،(9)

 

10 وَ إِذَا الْجِبالُ نُسِفَتْ

و آنگاه كه كوهها از جا كنده و متلاشى شوند،(10)

 

11 وَ إِذَا الرُّسُلُ أُقِّتَتْ

و در آن هنگام كه براى پيامبران (به منظور اداى شهادت) تعيين وقت شود.(11)

 

12 لاَِيِّ يَوْم أُجِّلَتْ

(اين شهادت) براى چه روزى به تأخير افتاده؟!(12)

 

13 لِيَوْمِ الْفَصْلِ

براى روزجدايى (حق از باطل).(13)

 

14 وَ ما أَدْراكَ ما يَوْمُ الْفَصْلِ

تو چه مى دانى روز جدايى چيست؟!(14)

 

15 وَيْلٌ يَوْمَئِذ لِلْمُكَذِّبِينَ

واى در آن روز بر تكذيب كنندگان.(15)

 

16 أَ لَمْ نُهْلِكِ الأَوَّلِينَ

آيا ما اقوام (مجرم) نخستين را هلاك نكرديم؟!(16)

 

17 ثُمَّ نُتْبِعُهُمُ الآخِرِينَ

سپس ديگران را به دنبال آنها مى فرستيم.(17)

 

18 كَذالِكَ نَفْعَلُ بِالْمُجْرِمِينَ

(آرى) اين گونه با مجرمان رفتار مى كنيم.(18)

 

19 وَيْلٌ يَوْمَئِذ لِلْمُكَذِّبِينَ

واى در آن روز بر تكذيب كنندگان!(19)

 

20 أَ لَمْ نَخْلُقْكُمْ مِنْ ماء مَهِين

آيا شما را از آبى پست و ناچيز نيافريديم،(20)

 

21 فَجَعَلْناهُ فِي قَرار مَكِين

سپس آن را در قرارگاهى مطمئن و آماده [= رحم] قرار داديم،(21)

 

22 إِلى قَدَر مَعْلُوم

تا مدّتى معيّن؟!(22)

 

23 فَقَدَرْنا فَنِعْمَ الْقادِرُونَ

ما توانايى بر اين كار داشتيم، پس ما چه تواناى خوبى هستيم (و امر معاد براى ما آسان است).(23)

 

24 وَيْلٌ يَوْمَئِذ لِلْمُكَذِّبِينَ

واى در آن روز بر تكذيب كنندگان!(24)

 

25 أَ لَمْ نَجْعَلِ الأَرْضَ كِفاتاً

آيا زمين را مركز اجتماع (شما) قرار نداديم،(25)

 

26 أَحْياءً وَ أَمْواتاً

براى زندگان و مردگان (شما)؟!(26)

 

27 وَ جَعَلْنا فِيها رَواسِيَ شامِخات وَ أَسْقَيْناكُمْ ماءً فُراتاً

و در آن كوههاى استوار و بلندى قرار داديم، و آبى گوارا به شما نوشانديم.(27)

 

28 وَيْلٌ يَوْمَئِذ لِلْمُكَذِّبِينَ

واى در آن روز بر تكذيب كنندگان! (28)

 

29 انْطَلِقُوا إِلى ما كُنْتُمْ بِهِ تُكَذِّبُونَ

(در آن روز به آنها گفته مى شود:) بى درنگ،به سوى همان چيزى كه پيوسته آن را تكذيب مى كرديد برويد. (29)

 

30 انْطَلِقُوا إِلى ظِلّ ذِي ثَلاثِ شُعَب

برويد به سوى سايه اى سه شاخه (از دودهاى خفقان بار و آتش زا)!(30)

 

31 لا ظَلِيل وَ لا يُغْنِي مِنَ اللَّهَبِ

سايه اى كه نه آرام بخش است و نه ازشعله هاى آتش جلوگيرى مى كند!(31)

 

32 إِنَّها تَرْمِي بِشَرَر كَالْقَصْرِ

شراره هايى (از خود) پرتاب مى كند به بزرگى يك كاخ!(32)

 

33 كَأَنَّهُ جِمالَتٌ صُفْرٌ

گويى (در سرعت و كثرت) همچون شتران زرد رنگى هستند (كه به هر سو پراكنده مى شوند)!(33)

 

34 وَيْلٌ يَوْمَئِذ لِلْمُكَذِّبِينَ

واى در آن روز بر تكذيب كنندگان!(34)

 

35 هذا يَوْمُ لا يَنْطِقُونَ

امروز روزى است كه سخن نمى گويند (و قادر بر دفاع از خويشتن نيستند)،(35)

 

36 وَ لا يُؤْذَنُ لَهُمْ فَيَعْتَذِرُونَ

و به آنها اجازه داده نمى شود تا عذرخواهى كنند.(36)

 

37 وَيْلٌ يَوْمَئِذ لِلْمُكَذِّبِينَ

واى در آن روز بر تكذيب كنندگان!(37)

 

38 هذا يَوْمُ الْفَصْلِ جَمَعْناكُمْ وَ الأَوَّلِينَ

(و به آنها گفته مى شود:) امروز، روز جدايى (حق از باطل) است كه شما و پيشينيان را در آن جمع كرده ايم.(38)

 

39 فَإِنْ كانَ لَكُمْ كَيْدٌ فَكِيدُونِ

اگر چاره اى در برابر من (براى فرار از چنگال مجازات) داريد انجام دهيد!(39)

 

40 وَيْلٌ يَوْمَئِذ لِلْمُكَذِّبِينَ

واى در آن روز بر تكذيب كنندگان!(40)

 

41 إِنَّ الْمُتَّقِينَ فِي ظِلال وَ عُيُون

(در آن روز) پرهيزگاران در سايه هاى (درختان بهشتى) و در ميان چشمه ها قرار دارند،(41)

 

42 وَ فَواكِهَ مِمّا يَشْتَهُونَ

و ميوه هايى از آنچه مايل باشند.(42)

 

43 كُلُوا وَ اشْرَبُوا هَنِيئاً بِما كُنْتُمْ تَعْمَلُونَ

بخوريد و بنوشيد گوارا باد بر شما، اينها در برابراعمالى است كه انجام مى داديد!(43)

 

44 إِنّا كَذالِكَ نَجْزِي الْمُحْسِنِينَ

ما اين گونه نيكوكاران را پاداش مى دهيم.(44)

 

45 وَيْلٌ يَوْمَئِذ لِلْمُكَذِّبِينَ

واى در آن روز بر تكذيب كنندگان!(45)

 

46 كُلُوا وَ تَمَتَّعُوا قَلِيلاً إِنَّكُمْ مُجْرِمُونَ

(و به مجرمان بگو:) بخوريد و اندكى (از زندگى دنيا) بهره گيريد (ولى بدانيد عذاب الهى در انتظار شماست) چرا كه شما مجرميد.(46)

 

47 وَيْلٌ يَوْمَئِذ لِلْمُكَذِّبِينَ

واى در آن روز بر تكذيب كنندگان!(47)

 

48 وَ إِذا قِيلَ لَهُمُ ارْكَعُوا لا يَرْكَعُونَ

و هنگامى كه به آنها گفته شود ركوع كنيد ركوع نمى كنند.(48)

 

49 وَيْلٌ يَوْمَئِذ لِلْمُكَذِّبِينَ

واى در آن روز بر تكذيب كنندگان!(49)

 

50 فَبِأَيِّ حَدِيث بَعْدَهُ يُؤْمِنُونَ

(و اگر آنها به اين قرآن ايمان نمى آورند) پس به كدام سخن بعد از آن ايمان مى آورند؟!(50)
